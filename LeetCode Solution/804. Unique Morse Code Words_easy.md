var uniqueMorseRepresentations = function(words) {
	let mos = [".-", "-...", "-.-.", "-..", ".", "..-.", "--.", "....", "..", ".---", "-.-", ".-..", "--", "-.", "---", ".--.", "--.-", ".-.", "...", "-", "..-", "...-", ".--", "-..-", "-.--", "--.."];
	let wtm = new Set();
	for(let v of words) {
		let m = '';
		for(let w of v) {
			m += mos[w.charCodeAt(0) - 'a'.charCodeAt(0)];
		}
		console.log(m)
		wtm.add(m);
	}
	return wtm.size;
};

为了方便，所有26个英文字母对应摩尔斯密码表如下：
```
[".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."]
```
给定一个单词列表，每个单词可以写成每个字母对应摩尔斯密码的组合。例如，"cab" 可以写成 "-.-.-....-"，(即 "-.-." + "-..." + ".-"字符串的结合)。我们将这样一个连接过程称作单词翻译。

返回我们可以获得所有词不同单词翻译的数量。<br>
